⚙️ Setup — Import CSV into SQL Server
-- Step 1: Create the table
CREATE TABLE user_events (
    event_id        INT           PRIMARY KEY,
    user_id         INT           NOT NULL,
    event_type      VARCHAR(50)   NOT NULL,
    event_date      DATETIME2     NOT NULL,
    product_id      INT           NOT NULL,
    amount          FLOAT         NULL,
    traffic_source  VARCHAR(30)   NOT NULL
);

-- Step 2: Import CSV using BULK INSERT
BULK INSERT user_events
FROM 'C:\path\to\user_events.csv'
WITH (
    FIRSTROW = 2,
    FIELDTERMINATOR = ',',
    ROWTERMINATOR = '\n',
    TABLOCK
);

Q1
How many total events are in the table? Also show the count for each event type.
COUNTGROUP BY
✦ Solution
-- Total events
SELECT COUNT(*) AS total_events
FROM user_events;

-- Events per type
SELECT
    event_type,
    COUNT(*) AS event_count
FROM user_events
GROUP BY event_type
ORDER BY event_count DESC;
💡 GROUP BY partitions rows into groups — one per unique event_type — so COUNT(*) runs independently on each group.
Q2
Find all unique users who made at least one purchase.
DISTINCTWHERE
✦ Solution
SELECT DISTINCT user_id
FROM user_events
WHERE event_type = 'purchase'
ORDER BY user_id;
💡 DISTINCT removes duplicate user_ids so each buyer appears once even if they purchased multiple times.
Q3
What is the total revenue, average order value, and number of orders?
SUMAVGRevenue
✦ Solution
SELECT
    COUNT(*)                    AS total_orders,
    SUM(amount)                 AS total_revenue,
    ROUND(AVG(amount), 2)       AS avg_order_value,
    MIN(amount)                 AS min_order,
    MAX(amount)                 AS max_order
FROM user_events
WHERE event_type = 'purchase';
💡 Filter to purchases only before aggregating — otherwise AVG(amount) is skewed by the many NULLs from non-purchase events.
Q4
How many events came from each traffic source?
GROUP BYTraffic
✦ Solution
SELECT
    traffic_source,
    COUNT(*) AS total_events,
    COUNT(DISTINCT user_id) AS unique_users
FROM user_events
GROUP BY traffic_source
ORDER BY total_events DESC;
💡 COUNT(DISTINCT user_id) tells you how many unique users came from each channel — different from total events because one user can generate many events.
Q5
List the top 5 products by number of page views.
TOP / LIMITProducts
✦ Solution
SELECT TOP 5
    product_id,
    COUNT(*) AS page_views
FROM user_events
WHERE event_type = 'page_view'
GROUP BY product_id
ORDER BY page_views DESC;
💡 In SQL Server use TOP N; in MySQL/PostgreSQL use LIMIT N. Always pair with ORDER BY, otherwise TOP returns arbitrary rows.

Q6
Build a conversion funnel: show how many unique users reached each stage (page_view → add_to_cart → checkout_start → payment_info → purchase).
FunnelCASE WHEN
✦ Solution
SELECT
    COUNT(DISTINCT CASE WHEN event_type = 'page_view'       THEN user_id END) AS stage_1_views,
    COUNT(DISTINCT CASE WHEN event_type = 'add_to_cart'     THEN user_id END) AS stage_2_cart,
    COUNT(DISTINCT CASE WHEN event_type = 'checkout_start'  THEN user_id END) AS stage_3_checkout,
    COUNT(DISTINCT CASE WHEN event_type = 'payment_info'    THEN user_id END) AS stage_4_payment,
    COUNT(DISTINCT CASE WHEN event_type = 'purchase'        THEN user_id END) AS stage_5_purchase
FROM user_events;
💡 CASE WHEN inside COUNT(DISTINCT) is the classic funnel pattern — it conditionally counts users only when they hit each stage. The drop-off between stages reveals where users abandon.
Q7
Which traffic source generates the most revenue? Show revenue and conversion rate (purchasers / visitors) per source.
RevenueHAVINGTraffic
✦ Solution
SELECT
    traffic_source,
    COUNT(DISTINCT user_id)              AS total_visitors,
    COUNT(DISTINCT CASE WHEN event_type = 'purchase'
          THEN user_id END)               AS buyers,
    ROUND(SUM(CASE WHEN event_type = 'purchase'
          THEN amount ELSE 0 END), 2)      AS revenue,
    ROUND(
        100.0 * COUNT(DISTINCT CASE WHEN event_type = 'purchase' THEN user_id END)
        / COUNT(DISTINCT user_id), 2
    )                                      AS conversion_pct
FROM user_events
GROUP BY traffic_source
ORDER BY revenue DESC;
💡 Multiplying by 100.0 (not 100) forces decimal division in SQL Server. Integer / integer = integer (truncated)!
Q8
Find users who added a product to cart but never completed a purchase (abandoned cart users).
NOT INSubqueryCart Abandonment
✦ Solution
-- Method 1: NOT IN with subquery
SELECT DISTINCT user_id AS abandoned_cart_user
FROM user_events
WHERE event_type = 'add_to_cart'
  AND user_id NOT IN (
      SELECT DISTINCT user_id
      FROM user_events
      WHERE event_type = 'purchase'
  );

-- Method 2: LEFT JOIN (more performant on large tables)
SELECT DISTINCT c.user_id
FROM user_events c
LEFT JOIN user_events p
    ON c.user_id = p.user_id AND p.event_type = 'purchase'
WHERE c.event_type = 'add_to_cart'
  AND p.user_id IS NULL;
💡 The LEFT JOIN approach is faster on large tables than NOT IN because it avoids rescanning the subquery for every row. p.user_id IS NULL means "no matching purchase row was found."
Q9
Show daily revenue and the 7-day rolling average revenue.
Window FunctionAVG OVERTime Series
✦ Solution
WITH daily_revenue AS (
    SELECT
        CAST(event_date AS DATE)  AS sale_date,
        ROUND(SUM(amount), 2)      AS daily_rev
    FROM user_events
    WHERE event_type = 'purchase'
    GROUP BY CAST(event_date AS DATE)
)
SELECT
    sale_date,
    daily_rev,
    ROUND(AVG(daily_rev) OVER (
        ORDER BY sale_date
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    ), 2) AS rolling_7day_avg
FROM daily_revenue
ORDER BY sale_date;
💡 ROWS BETWEEN 6 PRECEDING AND CURRENT ROW defines a sliding window of 7 rows (today + 6 prior days). Window functions don't collapse rows like GROUP BY does.
Q10
For each product, calculate the view-to-purchase conversion rate.
CASE WHENConversionProducts
✦ Solution
SELECT
    product_id,
    COUNT(DISTINCT CASE WHEN event_type = 'page_view'  THEN user_id END) AS viewers,
    COUNT(DISTINCT CASE WHEN event_type = 'purchase'   THEN user_id END) AS buyers,
    ROUND(
        100.0 *
        COUNT(DISTINCT CASE WHEN event_type = 'purchase' THEN user_id END)
        / NULLIF(COUNT(DISTINCT CASE WHEN event_type = 'page_view' THEN user_id END), 0)
    , 2)                                                                   AS conversion_pct
FROM user_events
GROUP BY product_id
ORDER BY conversion_pct DESC;
💡 NULLIF(expr, 0) prevents divide-by-zero errors by turning 0 into NULL (division by NULL returns NULL gracefully instead of crashing).

Q11
Rank users by total revenue spent using DENSE_RANK(). Show top 10 customers.
DENSE_RANKWindow FunctionCTE
✦ Solution
WITH user_spend AS (
    SELECT
        user_id,
        COUNT(*)            AS total_purchases,
        ROUND(SUM(amount), 2) AS total_spent
    FROM user_events
    WHERE event_type = 'purchase'
    GROUP BY user_id
),
ranked AS (
    SELECT
        user_id,
        total_purchases,
        total_spent,
        DENSE_RANK() OVER (ORDER BY total_spent DESC) AS spend_rank
    FROM user_spend
)
SELECT * FROM ranked WHERE spend_rank <= 10;
💡 DENSE_RANK vs RANK: if two users tie at rank 1, DENSE_RANK gives the next user rank 2 (no gaps), while RANK skips to rank 3. Use DENSE_RANK for customer tiers.
Q12
Calculate session-level metrics: for each user, find their average time between page_view and purchase (in minutes) for completed journeys.
DATEDIFFSelf JOINSessions
✦ Solution
WITH journeys AS (
    SELECT
        pv.user_id,
        pv.product_id,
        pv.event_date                           AS view_time,
        p.event_date                            AS purchase_time,
        DATEDIFF(MINUTE, pv.event_date, p.event_date) AS minutes_to_purchase
    FROM user_events pv
    INNER JOIN user_events p
        ON  pv.user_id    = p.user_id
        AND pv.product_id = p.product_id
        AND pv.event_type = 'page_view'
        AND  p.event_type = 'purchase'
        AND  p.event_date > pv.event_date
)
SELECT
    user_id,
    COUNT(*)                            AS purchases,
    ROUND(AVG(1.0 * minutes_to_purchase), 1) AS avg_minutes_to_buy
FROM journeys
GROUP BY user_id
ORDER BY avg_minutes_to_buy;
💡 A self-join joins a table to itself using aliases (pv = page view, p = purchase). The condition p.event_date > pv.event_date ensures we only match events in the correct chronological order.
Q13
Identify "high-value" users: those whose total spend is above the average spend of all buyers. Show their spend and percentile.
PERCENT_RANKCTESegmentation
✦ Solution
WITH buyer_spend AS (
    SELECT
        user_id,
        ROUND(SUM(amount), 2) AS total_spent
    FROM user_events
    WHERE event_type = 'purchase'
    GROUP BY user_id
),
percentiles AS (
    SELECT
        user_id,
        total_spent,
        ROUND(PERCENT_RANK() OVER (ORDER BY total_spent) * 100, 1) AS percentile
    FROM buyer_spend
)
SELECT *
FROM percentiles
WHERE total_spent > (SELECT AVG(total_spent) FROM buyer_spend)
ORDER BY total_spent DESC;
💡 PERCENT_RANK() returns a value between 0 and 1 indicating where a row falls in its partition. A percentile of 90 means "this user spends more than 90% of all buyers."
Q14
Find the most popular product per traffic source (the product with the most page views for each source).
ROW_NUMBERPartitioningProducts
✦ Solution
WITH source_product_views AS (
    SELECT
        traffic_source,
        product_id,
        COUNT(*) AS view_count
    FROM user_events
    WHERE event_type = 'page_view'
    GROUP BY traffic_source, product_id
),
ranked AS (
    SELECT
        traffic_source,
        product_id,
        view_count,
        ROW_NUMBER() OVER (
            PARTITION BY traffic_source
            ORDER BY view_count DESC
        ) AS rn
    FROM source_product_views
)
SELECT traffic_source, product_id, view_count
FROM ranked
WHERE rn = 1
ORDER BY traffic_source;
💡 PARTITION BY is the key — it resets the row number for each traffic_source group. Filtering WHERE rn = 1 gives only the top-ranked product per group (like a per-group TOP 1).
Q15
Cohort analysis: for users who first visited in each week, what % eventually made a purchase?
CTEDATEPARTCohort
✦ Solution
WITH first_visit AS (
    -- Find each user's first visit week
    SELECT
        user_id,
        DATEPART(YEAR,  MIN(event_date))  AS cohort_year,
        DATEPART(WEEK,  MIN(event_date))  AS cohort_week
    FROM user_events
    WHERE event_type = 'page_view'
    GROUP BY user_id
),
buyers AS (
    SELECT DISTINCT user_id FROM user_events WHERE event_type = 'purchase'
)
SELECT
    f.cohort_year,
    f.cohort_week,
    COUNT(f.user_id)                         AS cohort_size,
    COUNT(b.user_id)                         AS converted,
    ROUND(100.0 * COUNT(b.user_id) / COUNT(f.user_id), 2) AS conversion_pct
FROM first_visit f
LEFT JOIN buyers b ON f.user_id = b.user_id
GROUP BY f.cohort_year, f.cohort_week
ORDER BY f.cohort_year, f.cohort_week;
💡 Cohort analysis groups users by when they first arrived, then tracks behavior over time. LEFT JOIN to buyers ensures users with no purchase still appear (counted as 0 converted). This pattern is used extensively in product analytics.
